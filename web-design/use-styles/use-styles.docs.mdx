---
description: 'A hook that generates jss using styled component themes and a callback.'
---

```ts

/** import your theme here */
import {Theme} from './Theme';

import {DefaultTheme, useTheme, ThemeProvider} from 'styled-components';

const makeStyles = (theme: DefaultTheme) => ({
  view: {
    background: theme.colors.background,
    width: '100px',
    height: '100px'
  },
  title: {
    color: 'white',
  }
})

const StyledComponent = () => {
  const theme = useTheme();
  const styles = useStyles(makeStyles);
  return (
    <View style={styles.view}>
      <Title style={styles.color}>KiloLab</Title>
    </View>
  )
}

export const BasicUseStyles = () => {
  return (
    /** Pass your theme object to the styled components theme provider! */
    <ThemeProvider theme={Theme}>
      <StyledComponent />
    </ThemeProvider>
  );
};

```
